	ORG 1000H
MSJ 	DB "INGRESE UN NUMERO:"
FIN 	DB ?
SIES	DB "EL CARACTER ES UN NUMERO"
FINSI	DB ?
NOES	DB "EL CARACTER NO ES UN NUMERO"
FINNO	DB ?
	ORG 1500H
NUM 	DB ?

	ORG 3000H
ES_NUM: ;MOV CL, OFFSET[BX]
	SUB CL,30H
	JN NO
	SUB CL,10H
	JNZ NO
	MOV BX, OFFSET SIES
	MOV AL, OFFSET FINSI
	INT 7
	JMP FIN
NO:	MOV BX, OFFSET NOES
	MOV AL, OFFSET FINNO
	INT 7
	FINNO
	JMP FIN
FIN:	RET
	
	ORG 2000H
	MOV BX, OFFSET MSJ
	MOV AL, OFFSET FIN-OFFSET MSJ
	INT 7
	MOV BX, OFFSET NUM
	INT 6
	MOV AL, 1
	INT 7
	MOV CL, NUM
	CALL ES_NUM
	INT 0
	END
